#expr       : KEYWORD:var IDENTIFIER EQ expr
           | comp-expr ((KEYWORD:and|KEYWORD:or) comp-expr)*


#comp-expr  : not comp-expr
           | arith_expr ((EE|LT|GT|LTE|GTE) arith_expr)*

#arith-expr : term ((PLUS|MINUS) term)*


#term       : factor ((MUL|DIV) factor)*

#factor     : (PLUS|MINUS) factor
           | power

#power      : atom (POW factor)*

#atom       : INT
           | FLOAT
           | IDENTIFIER
           | LPAREN expr RPAREN
           | if-expr
           | for-expr
           | while-expr


#if-expr    : KEYWORD:if expr KEYWORD:then expr
           | (KEYWORD:elif expr KEYWORD:then expr)*
           | (KEYWORD:else expr)?

#for-expr   : KEYWORD:for IDENTIFIER EQ expr KEYWORD:to expr
             (KEYWORD:step expr)? KEYWORD:then expr

#while-expr : KEYWORD:while expr KEYWORD:then expr

clipCreation : videoCreation
             | photoCreation

videoCreation: KEYWORD:videoClip lparen filename, number, number, float rparen

photoCreation: keyword:photoClip lparen filename, number rparen

method: var dot methodcall

methodcall : concatenate_clip
           | resize
           | crop
           | add_text
           | addAudioFromFile
           | addAudioFromClip
           | writeVideo
           | create_gif
